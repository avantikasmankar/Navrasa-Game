    <script>

        const navrasa = [

            { sanskrit: 'Shringara', english: 'Love' },

            { sanskrit: 'Hasya', english: 'Laughter' },

            { sanskrit: 'Karuna', english: 'Compassion' },

            { sanskrit: 'Raudra', english: 'Anger' },

            { sanskrit: 'Veera', english: 'Heroism' },

            { sanskrit: 'Bhayanaka', english: 'Fear' },

            { sanskrit: 'Bibhatsa', english: 'Disgust' },

            { sanskrit: 'Adbhuta', english: 'Wonder' },

            { sanskrit: 'Shanta', english: 'Peace' }

        ];


        let isShuffling = false;


        function shuffleCards() {

            if (isShuffling) return;

            

            isShuffling = true;

            const btn = document.getElementById('shuffleBtn');

            const container = document.getElementById('cardsContainer');

            const result = document.getElementById('result');

            

            btn.textContent = 'Shuffling...';

            btn.disabled = true;

            

            setTimeout(() => {

                // Shuffle array and pick 3

                const shuffled = [...navrasa].sort(() => Math.random() - 0.5);

                const drawn = shuffled.slice(0, 3);

                

                // Clear container

                container.innerHTML = '';

                

                // Add drawn cards

                drawn.forEach(card => {

                    const cardElement = document.createElement('div');

                    cardElement.className = 'card';

                    cardElement.innerHTML = `

                        <div class="sanskrit">${card.sanskrit}</div>

                        <div class="english">${card.english}</div>

                    `;

                    container.appendChild(cardElement);

                });

                

                // Show result

                result.textContent = `Your emotional trilogy: ${drawn.map(card => card.english).join(' • ')}`;

                

                // Reset button

                btn.textContent = 'Shuffle Cards';

                btn.disabled = false;

                isShuffling = false;

            }, 800);

        }

    </script>
